  
།རྒྱ་གར་སྐད་དུ། ཨཱརྱ་ཏཱ་རཱ་དེ་བའི་སྟོ་ཏྲ། བོད་སྐད་དུ། འཕགས་མ་ལྷ་མོ་སྒྲོལ་མ་ལ་  
བསྟོད་པ། རྗེ་བཙུན་འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་ལ་ཕྱག་འཚལ་ལོ། །ལྷ་མོ་སྒྲོལ་མ་ལྷ་ཡི་ལྷ། །འཇིག་རྟེན་གསུམ་ན་མཆོག་གྱུར་པ། །ཐུགས་རྗེ་ཆེན་པོའི་དབང་གྱུར་པས། །ཁམས་གསུམ་འགྲོ་བ་སྒྲོལ་ལ་མཁས། །ཚོར་བ་བཏང་སྙོམས་ངོ་བོ་  
ཉིད༑ །མི་མཐུན་ཚོར་བ་བདེ་སྡུག་ལྡན། ཕྲག་དོག་འཇོམས་པའི་ཕྲིན་ལས་ཅན། །མཁའ་ལྟར་དངས་པ་བླ་ན་མེད། །གཟུང་བ་དང་ནི་འཛིན་པ་ཡི། །རྣམ་རྟོག་སྤངས་པའི་བདེ་བ་ཅན། །པདྨ་ཆུས་མི་ཆགས་པ་བཞིན། །འཁོར་བའི་སྐྱོན་གྱིས་མ་གོས་པ། །རང་བཞིན་གྱིས་  
དག་བླ་ན་མེད། །རྒོལ་བ་དག་ནི་ཚར་གཅོད་པ། །འགྲོ་བ་ཡིད་འོང་ཕུན་ཚོགས་པ། །སྐུ་མདོག་མཚན་དང་དཔེ་བྱད་ལྡན། །སྒྲོལ་མ་སྣ་ཚོགས་མདོག་ཅན་ནི། །འབྱུང་པོ་ལས་བྱུང་གདོན་ཆེན་དང། །འདུ་བ་ལས་བྱུང་ནད་རྣམས་དང། །བཀྲ་མི་ཤིས་ནི་སྣ་ཚོགས་དང། །མཚོན་དང་  
དུག་དང་རིག་སྔགས་དང། །ཕྲ་མེན་ལས་བྱུང་སྦྱོར་བ་ཡིས། །ཤིན་ཏུ་མི་བཟད་ཐབས་ལྡན་པའི། །སྟོབས་ནི་སྣ་ཚོགས་འཇོམས་པ་ངང། །སེང་གེ་གླང་པོ་སྟག་དང་སྤྲུལ། །ཆོམ་རྐུན་མེ་ཆུ་ཤ་ཟ་ཡི། །འཇིགས་པ་ཆེན་པོ་དེ་རྣམས་ལས། །སྐྱོབ་མཛད་དེ་ལ་བསྟོད་ཕྱག་འཚལ། །  
འཕགས་མ་ལྷ་མོ་སྒྲོལ་མ་ལ་བསྟོད་པ་རྫོགས་སོ།།
