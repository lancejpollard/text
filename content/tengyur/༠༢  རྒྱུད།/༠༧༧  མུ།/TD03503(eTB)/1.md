  
།རྒྱ་གར་སྐད་དུ། ཨཱརྱ་ཏཱ་རེ་དེ་བའི་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། ལྷ་མོ་སྒྲོལ་མའི་སྒྲུབ་ཐབས། ལྷ་མོ་སྒྲོལ་མ་ལ་ཕྱག་འཚལ་ལོ། །སྔར་བཤད་པའི་ཆོ་  
  
༄༅། །གས་བདུད་རྩིའི་དཀྱིལ་འཁོར་དེའི་དབུས་སུ་ཡི་གེ་སྡེ་ཚན་ལྔ་པའི་དང་པོའི་ས་བོན་ལ་གཉིས་པས་རྫོགས་པར་བྱས་ལ་ཟླ་བ་ཕྱེད་པ་དང་ཐིག་ལེས་བརྒྱན་པ་དེ་ཉིད་ལས་བྱུང་བའི་ལྷ་མོ་སྒྲོལ་མ་བསྒོམ་པར་བྱའོ། །སྐུ་མདོག་ལྗང་གུ་ཕྱག་གཉིས་པ། གཡས་ཀྱིས་མཆོག་  
སྦྱིན་པ༑ གཡོན་པས་ཨུཏྤ་ལ་བསྣམས་པ། པདྨ་དང་ཟླ་བ་ལ་བཞུགས་པ། རྒྱན་ཐམས་ཅད་ཀྱིས་བརྒྱན་པ་བསམ་པར་བྱའོ། དེ་ནས་ཕྱག་རྒྱ་བཅིང་བར་བྱ་སྟེ། ལག་པ་གཉིས་ཐལ་མོ་སྦྱར་ལ་མཛུབ་མོ་གཉིས་ཀྱིས་གུང་མོ་བཅིངས་ལ། བཅང་བ་ནི་ལྷ་མོ་སྒྲོལ་མའི་ཨུཏྤ་ལ་ཁ་  
བྱེ་བའི་ཕྱག་རྒྱའོ། །དེའི་འོག་ཏུ་སྔགས་བཟླས་པར་བྱ་སྟེ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སཱཧཱ། ལྷ་མོ་སྒྲོལ་མའི་སྒྲུབ་ཐབས་རྫོགས་སོ།།
