  
།རྒྱ་གར་སྐད་དུ། ཨཱ་ཡུ་པ་རི་རཀྵ་ནཱ་མ། བོད་སྐད་དུ། ཚེ་སྒྲུབ་པའི་གདམས་ངག་ཅེས་བྱ་བ། བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །བླ་མ་རྣམས་ལ་ཕྱག་འཚལ་ནས། །བཅུད་ལེན་ཞལ་གྱི་གདམས་པ་ཡིས། །ལུས་ངག་རྟག་ཏུ་བསྲུང་བར་བྱ། །དེ་མེད་

སོ་དང་སེན་མོ་ཟླ་དྲུག་གིས། །བརྗེས་འགྱུར་ལོ་ནི་སྟོང་ཐུབ་སྟེ། །རྟགས་ནི་མིག་རྣོ་སྟོབས་ལྡན་ནོ། །ཉེ་ཤིང་རྩ་བ་བླང་བྱས་ནས། །གྲིབ་སྐམ་ཞིབ་ཏུ་བཏགས་པར་བྱ། །འོ་མའམ་ཞོ་དང་སྦྱར་བྱས་ནས། །ཞོ་ནི་ལངས་ནས་སྐྲོག་པར་བྱ། །མར་སར་བླངས་ནས་ཐོ་རངས་བཟའ། །ཞག་ནི་  
བདུན་ནས་རྟགས་འབྱུང་སྟེ། །བདུན་གཉིས་གྱུར་ནས་ཉི་འོད་ལྡན། །བདུན་གསུམ་ཉི་ཤུ་རྩ་གཅིག་ནས། །འབྲས་བུ་ནང་སེལ་ལུས་ཡང་ངོ། །ལོ་དྲུག་ཏུ་ནི་བཟའ་བཏུང་གིས། །རྣམ་རྟོག་བྲལ་ཞིང་ཚེ་འགྲུབ་བོ། །ཐོ་རས་མུ་ཟི་སྲང་ཕྱེད་ནི། །ཡུངས་ཀར་མར་ཁུས་བཏུལ་ནས་བཟའ། །
