  
།རྒྱ་གར་སྐད་དུ། ས་ཧཱ་ཛ་ཨ་ནནྟ་ཨཱ་བཱ་ཡཾ་ནཱ་མ། བོད་སྐད་དུ། ལྷན་ཅིག་སྐྱེས་པའི་གདམས་པ་ཞེས་བྱ་བ། བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །སྙིང་གའི་ཁ་སྦྱོར་དབུས་སུ་ནི། །གཉིས་མེད་ཐིག་ལེ་ཡུངས་ཀར་  
ཚད༑ །དེ་ཡང་འོ་མའི་རྒྱ་མཚོ་ལྟར། །རིམ་གྱིས་རྒྱས་པ་བསམ་པར་བྱ། །བདག་དང་གཞན་གྱི་ལུས་ཉིད་ནི། །དྲི་མེད་བདུད་རྩིས་ཞོ་ལྟར་གང། །རླུང་ན་སེལ་ཞིང་སེམས་ཉིད་བཅིང། །མེ་དཧི་བདག་ནི་སྨྲ་བར་བྱེད། །དངོས་པོར་སེམས་གཏང་དངོས་མེད་རྟོགས། །  
ཀྱེ་ཧོ་མ་ཡེངས་རང་སེམས་ཟུང། །རྣལ་འབྱོར་གྲུབ་རྟགས་དངོས་མེད་པ། །ཕན་ཚུན་གཉིས་ཀྱི་འབྲེལ་པ་བཅད། །དངོས་དང་དངོས་མེད་རང་བཞིན་ལ། །མཐའ་གཉིས་མི་གནས་ལྷན་སྐྱེས་གྲུབ། །དངོས་པོ་ཐམས་ཅད་སྐྱེ་མེད་དེ། །བསྒྲིབས་པས་བདག་གིས་མ་རྟོགས་  
  
།སྣང་བ་ཐམས་ཅད་སེམས་ངོ་བོ། །སེམས་ཉིད་མ་གྲུབ་སེམས་ཉིད་དེ། །བདག་ཉིད་སྙིང་རྗེ་ཀུན་དུ་ཁྱབ། །ནམ་མཁའ་ལྟ་བུར་འགྱུར་བ་མེད། །གཉིས་ལ་མི་གནས་སེམས་ཉིད་ཡིན། །མེ་དཧི་བདག་ནི་སྨྲ་བར་བྱེད། །རང་རིག་མ་བཅོས་བྱ་བ་ཐོང། །འཛིན་མེད་དྲན་མེད་  
སེམས་ཉིད་དེ། །ཆད་པར་མི་འགྱུར་རང་རིག་རྟོགས། །རྟག་པར་མི་འགྱུར་སྒྲོ་འདོགས་བཅད། །སྟོང་པའི་རང་བཞིན་རྟག་ཆད་མེད། །མེ་དཧི་བདག་ནི་སྨྲ་བར་བྱེད། །ལྟ་བ་བསྒོམ་མེད་འབྲས་བུ་མེད། །སྣང་ལ་རང་བཞིན་དངོས་པོ་མེད། །རྣམ་རྟོག་སྤྲུལ་གྱི་མདུད་པ་ལྟར། །རང་གྲོལ་  
ཟུང་འཇུག་ཐ་སྙད་མེད། །དེ་ལྟར་བསྒོམས་པས་སྟོང་ཉིད་རྟོགས། །མེ་དཧི་བདག་ནི་སྨྲ་བར་བྱེད། །ལྷན་སྐྱེས་ཟུང་འཇུག་སྟོང་ཉིད་དེ། །རྣལ་འབྱོར་བདག་གིས་རྟོགས་པའོ། །ལྷན་སྐྱེས་ཟུང་འཇུག་གི་གདམས་པ་སློབ་དཔོན་མེ་དཧི་ནིས་མཛད་པ་རྫོགས་སོ།། །།
