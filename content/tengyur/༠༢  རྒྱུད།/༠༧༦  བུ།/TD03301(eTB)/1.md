ཡང་རུང་བར་སྟན་ཅི་ཡང་རུང་བ་ལ་ཉེ་བར་གནས་ལ་རྡོ་རྗེ་རྣལ་འབྱོར་མ་བསྒོམ་པ་བརྩམ་པར་བྱ་བའོ། །དེ་ལ་རང་གི་ལྟེ་བར་ཁ་བྱེ་བའི་པདྨ་ཁ་དོག་དཀར་པོ་བསྒོམ་པར་བྱ་བའོ། །དེའི་སྟེང་དུ་རྩིའི་ཁ་དོག་ལྟ་བུའི་ཆོས་འབྱུང་དབུས་ན་ཡི་གེ་ཧྲཱིཾ་གནས་པར་བསྒོམ་པར་བྱ་བའོ། །དེ་ལ་  
ཡང་སྟེང་དུ་རྡེ་བཙུན་རྡོ་རྗེ་རྣལ་འབྱོར་མའོ། །དེ་ལ་གནས་པའི་ཡི་གེ་ཧྲཱིཾ་ཡོངས་སུ་གྱུར་པ་ལས་སྐུ་མདོག་སེར་མོ་སྟ། རང་ཉིད་ཁོ་ན་རང་གི་གྲི་གུག་དང། ཕྱག་གཡོན་པ་གྲི་གུག་བསྣམས་པ། རང་གི་དབུའི་སྟེང་དུ་གནས་པ། ཕྱག་གཡས་བ་ནི་ཐོད་པ་དང་ལྡན་པའོ། །དཔུང་པ་གཡོན་  
པ་ནི་གྱེན་ལ་བརྐྱང་བའོ། །དཔུང་པ་གཡས་པ་ནི་འོག་ཏུ་སྨད་པའོ། །གོས་མེད་པ་ཞབས་གཡས་པ་རབ་ཏུ་བརྐྱང་པ། ཞབས་གཡོན་པ་ལེགས་པར་བསྐུམས་པར་བསྒོམ་པར་བྱ་བའོ། །ཨོཾ་འདུད་པ་ལ་ཀུན་ནས་བརྟེན་པའི་ཁྲག་གི་རྒྱུན་རང་ཉིད་ཀྱི་ཞལ་དུ་རབ་ཏུ་འབར་བ། གཞན་
