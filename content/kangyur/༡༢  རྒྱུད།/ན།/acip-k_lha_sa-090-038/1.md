བཅོམ་ལྡན་འདས་ཀྱི་འཁོར་རྣམ་པ་བཞི་དང་རིགས་དྲུག་གི་སེམས་ཅན་ཐམས་ཅད་ལ། སྤོས་དང། སྨན་དང། ཞལ་ཟས་ཀྱི་མཆོད་པ་ཆེན་པོ་བཤམས་ནས། བཅོམ་ལྡན་འདས་ལ་འདི་

རྩི་འབྱུང་བ་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན་ལ་སྙོམས་པར་ཞུགས་ཏེ། ཞལ་ཟས་ཀྱི་སྟེང་དུ་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་རཾ་དཀར་པོ་ལ་ཆོས་ཐམས་ཅད་རྡུལ་དང་བྲལ་བའོ་ཞེས་བྱ་བའི་སྒྲ་བྱུང་བས་  
ཞལ་ཟས་ཀྱི་དྲི་མ་དག་པར་གྱུར་ཏོ། །ཡང་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་ཨ་དཀར་པོ་བསམས་ཏེ། ཞལ་ཟས་ལྷའི་བདུད་རྩི་དང་ཡོ་བྱད་དུ་གྱུར་ཏོ། ཡང་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་
